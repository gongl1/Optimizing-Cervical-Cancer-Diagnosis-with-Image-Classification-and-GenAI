# Function to load and preprocess images with labels
def load_images(root_dir, classes):
    images = []
    labels = []
    for idx, cls in enumerate(classes):
        class_dir = os.path.join(root_dir, cls, "CROPPED")
        for img_name in os.listdir(class_dir):
            img_path = os.path.join(class_dir, img_name)
            img = load_img(img_path, target_size=(64, 64))
            img_array = img_to_array(img)
            images.append(img_array)
            labels.append(idx)
    images = np.array(images) / 255.0  # Normalize to [0, 1]
    labels = np.array(labels)
    return images, labels

x = layers.BatchNormalization()(x)
x = layers.Dropout(0.3)(x)
x = layers.Conv2D(512, 3, activation="relu", strides=2, padding="same")(x)
x = layers.BatchNormalization()(x)
x = layers.Flatten()(x)

x = layers.Dense(256, activation="relu")(x)



# Concatenate the flattened feature map with the label input
x = layers.Concatenate()([x, label_inputs])

# Instead of encoding an input directly to a single point in the latent space, the VAE encodes it to a distribution.
# layers.Dense(latent_dim): This creates a fully connected layer with latent_dim units (neurons).
# Each unit in this layer will output a value representing one dimension of the mean of the Gaussian distribution.
# (x): This applies the dense layer to the input tensor x, which is the output from the previous layers.
# The dense layer learns to map this input to the mean values of the latent variables.
z_mean = layers.Dense(latent_dim)(x)
z_log_var = layers.Dense(latent_dim)(x)
# The output is two vectors: z_mean and z_log_var, representing the mean and log variance of the latent variables.




# Sampling layer
def sampling(args):
    z_mean, z_log_var = args
    batch = tf.shape(z_mean)[0] # batch is the number of samples in the current batch.
    dim = tf.shape(z_mean)[1]   # dim is the dimensionality of the latent space (number of latent variables).
    epsilon = K.random_normal(shape=(batch, dim))
    return z_mean + K.exp(0.5 * z_log_var) * epsilon

z = layers.Lambda(sampling)([z_mean, z_log_var])
# The sampling layer generates a latent variable z from z_mean and z_log_var using the reparameterization trick.
# This ensures that the gradient can backpropagate through the sampling process during training.



# Decoder
decoder_inputs = tf.keras.Input(shape=(latent_dim + len(classes),))

x = layers.Dense(16 * 16 * 512, activation="relu")(decoder_inputs)
x = layers.Reshape((16, 16, 512))(x)
x = layers.Conv2DTranspose(512, 3, activation="relu", strides=2, padding="same")(x)  # 32x32
x = layers.BatchNormalization()(x)
x = layers.Conv2DTranspose(256, 3, activation="relu", strides=2, padding="same")(x)  # 64x64
x = layers.BatchNormalization()(x)
x = layers.Conv2DTranspose(128, 3, activation="relu", padding="same")(x)
x = layers.BatchNormalization()(x)
x = layers.Conv2DTranspose(64, 3, activation="relu", padding="same")(x)
x = layers.BatchNormalization()(x)
decoder_outputs = layers.Conv2DTranspose(3, 3, activation="sigmoid", padding="same")(x)  # 64x64
# The decoder part of the VAE maps the latent variable z back to the image space.



# VAE Model
encoder = models.Model([encoder_inputs, label_inputs], [z_mean, z_log_var, z], name="encoder")
# encoder = models.Model(encoder_inputs, [z_mean, z_log_var, z], name="encoder")
decoder = models.Model(decoder_inputs, decoder_outputs, name="decoder")

z_conditional = layers.Concatenate()([z, label_inputs])
vae_outputs = decoder(z_conditional)

# The VAE model is created by combining the encoder and decoder models.
# The encoder produces the latent variables z_mean and z_log_var, and samples z.
# The decoder takes z and generates the reconstructed image.





# ?! This encourages the generated image to not only match the pixel values of the input image but also to have similar high-level features, leading to more visually appealing and realistic results.
# ?! By integrating perceptual loss, the model not only focuses on minimizing pixel-wise differences but also ensures that the generated images have similar high-level features to the input images.
# This leads to better quality and more realistic generated images.

# Perceptual loss using VGG16
vgg = VGG16(weights='imagenet', include_top=False, input_shape=(64, 64, 3))
vgg.trainable = False

def perceptual_loss(y_true, y_pred):
    y_true_vgg = vgg(y_true)
    y_pred_vgg = vgg(y_pred)
    return tf.reduce_mean(tf.square(y_true_vgg - y_pred_vgg))


# ?! By combining these loss components, the model learns to generate high-quality images that closely resemble the input images while maintaining a well-structured latent space.

# Custom layer for VAE loss
# The loss function combines three components: the reconstruction loss, the perceptual loss, and the KL divergence loss.

# The custom loss layer is defined by the VAELossLayer class, which inherits from tf.keras.layers.Layer.
class VAELossLayer(layers.Layer):
    def __init__(self, **kwargs):
        super(VAELossLayer, self).__init__(**kwargs)

    def vae_loss(self, encoder_inputs, vae_outputs, z_mean, z_log_var):
        print(f"encoder_inputs shape: {encoder_inputs.shape}")
        print(f"vae_outputs shape: {vae_outputs.shape}")
        print(f"z_mean shape: {z_mean.shape}")
        print(f"z_log_var shape: {z_log_var.shape}")

        reconstruction_loss = tf.reduce_mean(tf.square(K.flatten(encoder_inputs) - K.flatten(vae_outputs)))
        reconstruction_loss *= 64 * 64 * 3

        # Perceptual Loss: Computes the difference between the VGG16 features of the original and reconstructed images.
        # This loss helps the model to focus on high-level features and improves the visual quality of the generated images.
        perceptual_loss_value = perceptual_loss(encoder_inputs, vae_outputs)

        # KL Divergence Loss: Measures how the learned latent distribution deviates from the standard normal distribution.
        # This ensures that the latent space is well-behaved.
        kl_loss = 1 + z_log_var - K.square(z_mean) - K.exp(z_log_var)
        kl_loss = K.sum(kl_loss, axis=-1)
        kl_loss *= -0.5
        return K.mean(reconstruction_loss + perceptual_loss_value + kl_loss)

    def call(self, inputs):
        encoder_inputs = inputs[0]
        vae_outputs = inputs[1]
        z_mean = inputs[2]
        z_log_var = inputs[3]

        # Print shapes for debugging
        print(f"encoder_inputs shape in call: {K.int_shape(encoder_inputs)}")
        print(f"vae_outputs shape in call: {K.int_shape(vae_outputs)}")
        print(f"z_mean shape in call: {K.int_shape(z_mean)}")
        print(f"z_log_var shape in call: {K.int_shape(z_log_var)}")

        loss = self.vae_loss(encoder_inputs, vae_outputs, z_mean, z_log_var)
        self.add_loss(loss)
        return vae_outputs

    # compute_output_shape: Defines the output shape of the custom layer.
    # In this case, it is set to the shape of the VAE outputs.
    def compute_output_shape(self, input_shape):
        return input_shape[1]


# The final VAE model is created by adding the custom VAE loss layer to the encoder inputs and the decoder outputs.
# The model is then compiled with the Adam optimizer.
# The VAELossLayer combines the encoder_inputs, vae_outputs, z_mean, and z_log_var to compute the loss and include it in the model's computation graph.
outputs = VAELossLayer()([encoder_inputs, vae_outputs, z_mean, z_log_var])
vae = models.Model([encoder_inputs, label_inputs], outputs, name="vae")

vae.compile(optimizer=tf.keras.optimizers.Adam(learning_rate=0.0005))
vae.summary()

def generate_new_images(decoder, n=10, class_label=None):
    random_latent_vectors = np.random.normal(size=(n, latent_dim))
    if class_label is not None:
        conditional_input = np.concatenate([random_latent_vectors, class_label], axis=1)
        generated_images = decoder.predict(conditional_input)
    else:
        generated_images = decoder.predict(random_latent_vectors)
    return generated_images

# Central Value (5): This value enhances the intensity of the central pixel. By multiplying the central pixel's intensity by 5, the kernel emphasizes the central pixel.
# Surrounding Values (-1): These values subtract the intensity of the neighboring pixels from the central pixel's intensity.
# This operation effectively highlights the edges and transitions between different regions of the image, making the image appear sharper.
def sharpen_image(image):
